#   Write A program to perform Arithmetic Operation .

# Algorithm :-
  1. START
  2. Declaration of Variable i.e., a,b,c
  3. Perform Addition
  4. Display Addition
  5. Perform Subtraction
  6. Display Subtraction
  7. Perform Multiplication
  8. Display Multiplication
  9. Perform Division
  10. Display Division
  11. Perform Remainder
  12. Display Remainder
  13. STOP 


// Write A program to perform Arithmetic Operation  

#include<stdio.h>    //header file 
void main(){               //main function 
  int a=7,b=5,c;           //Declaration 
  c = a + b;                          //Operation  
  printf("\n Addition = %d ",c);      //Output Statement  
  c = a - b; 
  printf("\n Subtraction = %d ",c);
  c = a * b; 
  printf("\n Multiplication = %d ",c);
  c = a / b; 
  printf("\n Division = %d ",c);
  c = a % b; 
  printf("\n Remainder = %d ",c);
}



                            
   Output :                
                        
       Addition =   12  

       Subtraction =    2  

       Multiplication =    35  

       Division =    1  

       Remainder =    2

#   Write A program to perform addition of 2 numbers if first number is smaller than second otherwise perform subtraction . 


// Write A program to perform addition of 2 numbers if first number is smaller than second otherwise perform subtraction .

#include<stdio.h> 
void main()
{
  int num1, num2, result;
  printf("\n Enter any two number : \n ");
  scanf("%d%d", &num1, &num2);
  if (num1 < num2)
  {
    printf(" \n \t %d is Smaller than %d hence ,\n", num1, num2);
    result = num1 + num2;
    printf("\n \t Addition of %d + %d = %d \n ", num1, num2, result);
  }
  else
  {
    printf(" \n \t %d is Greater  than %d hence ,\n", num1, num2);
    result = num1 - num2;
    printf("\n \t Subtraction  of %d - %d =  %d \n ", num1, num2, result);
  }
}



                            

  Output :

  CASE First:-

      Enter any two  numbers  : 
  
       10   
  
       20   
  
      
         
       10  is Smaller than  20   hence , 
  
      Addition of   10    +  20   =  30  
         

  CASE Second :- 

      Enter any two  numbers  : 
  
       30   
  
       10   
  
      
         
       30  is Greater  than  10   hence , 
  
      Subtraction  of   30   -  10   =  20
