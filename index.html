#   Write A program to perform Arithmetic Operation .

# Algorithm :-
  1. START
  2. Declaration of Variable i.e., a,b,c
  3. Perform Addition
  4. Display Addition
  5. Perform Subtraction
  6. Display Subtraction
  7. Perform Multiplication
  8. Display Multiplication
  9. Perform Division
  10. Display Division
  11. Perform Remainder
  12. Display Remainder
  13. STOP 


// Write A program to perform Arithmetic Operation  

#include<stdio.h>    //header file 
void main(){               //main function 
  int a=7,b=5,c;           //Declaration 
  c = a + b;                          //Operation  
  printf("\n Addition = %d ",c);      //Output Statement  
  c = a - b; 
  printf("\n Subtraction = %d ",c);
  c = a * b; 
  printf("\n Multiplication = %d ",c);
  c = a / b; 
  printf("\n Division = %d ",c);
  c = a % b; 
  printf("\n Remainder = %d ",c);
}



                            
   Output :                
                        
       Addition =   12  

       Subtraction =    2  

       Multiplication =    35  

       Division =    1  

       Remainder =    2

#   Write A program to calculate simple interest using runtime initialization . 


// Write A program to calculate simple interest using runtime initialization .

#include<stdio.h>   
void main(){             
  int p,r,n;          
  float si;
    printf("\n Enter Principal Amount : ");
    scanf("%d",&p);
    printf("\n Enter Rate of Interest : ");
    scanf("%d",&r);
    printf("\n Enter no. of year : ");
    scanf("%d",&n);
    si = (p*r*n )/ 100;
    printf("\n Simple Interest = %f",si); 
  }



                            
 
  Output :

  Enter Principal Amount :  10000  

  Enter Rate of Interest :  3  

  Enter no. of year :  5  

  Simple Interest =  1500.000000

// Write A program to calculate simple interest using runtime initialization .

#include<stdio.h>   
void main(){             
  float f,c;
  printf("Enter the temperature in fahrenheit:");
  scanf("%f",&f);

}

#   Write A program to calculate total & percentage of 5 subject mark. 


// Write A program to calculate total & percentage of 5 subject mark .

#include<stdio.h>   
void main(){             
int m1,m2,m3,m4,m5,total;
float percentage;
printf("\n Enter the marks of 5 subjects:");
scanf("%d%d%d%d%d",&m1,&m2,&m3,&m4,&m5);
total = m1+m2+m3+m4+m5;
percentage = total/5;
printf("\n Total = %d",total);
printf("\n Percentage = %f",percentage);

}



                            

  Output :

  Enter the marks of five subjects:  12 23 34 45 56  

     Total =  170  
  
     Percentage =  34.00 %
